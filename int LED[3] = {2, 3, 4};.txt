int LED[3] = {2, 3, 4};

void setup() {
  for (int i = 0; i < 3; i++) {
    pinMode(LED[i], OUTPUT);
  }
}

void loop() {
  for (int i = 0; i < 3; i++) {
    digitalWrite(LED[i], HIGH);
  }
  delay(1000);

  for (int i = 0; i < 3; i++) {
    digitalWrite(LED[i], LOW);
  }
  delay(1000);
}
