int ledPins[] = {2, 3, 4, 5, 6};
int n = 5;

void setup() {
  for (int i = 0; i < n; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
}

void loop() {
  // Forward
  for (int i = 0; i < n; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(300);
    digitalWrite(ledPins[i], LOW);
  }

  // Reverse
  for (int i = n - 1; i >= 0; i--) {
    digitalWrite(ledPins[i], HIGH);
    delay(300);
    digitalWrite(ledPins[i], LOW);
  }
}
